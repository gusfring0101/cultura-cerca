import React from "react";
import { Card, CardContent, CardHeader, CardTitle } from "@/components/ui/card";
import { Button } from "@/components/ui/button";
import { Label } from "@/components/ui/label";
import { Slider } from "@/components/ui/slider";
import { Badge } from "@/components/ui/badge";
import { MapPin, Euro, Heart, Search } from "lucide-react";

const culturalInterests = [
  "Museos", "Galerías de Arte", "Teatro", "Música en Vivo", "Exposiciones",
  "Bibliotecas", "Centros Culturales", "Cine", "Ópera", "Danza"
];

export default function FilterPanel({ filters, onFiltersChange, onSearch }) {
  const handleDistanceChange = (value) => {
    onFiltersChange({ ...filters, maxDistance: value[0] });
  };

  const handleBudgetChange = (value) => {
    onFiltersChange({ ...filters, maxBudget: value[0] });
  };

  const toggleInterest = (interest) => {
    const interests = filters.interests || [];
    const newInterests = interests.includes(interest)
      ? interests.filter((i) => i !== interest)
      : [...interests, interest];

    onFiltersChange({ ...filters, interests: newInterests });
  };

  return (
    <Card className="w-full h-fit shadow-xl border-0 bg-white/80 backdrop-blur-sm">
      <CardHeader className="pb-4">
        <CardTitle className="text-xl font-bold text-slate-800 flex items-center gap-2">
          <Search className="w-5 h-5 text-indigo-600" />
          Encuentra tu cultura ideal
        </CardTitle>
      </CardHeader>

      <CardContent className="space-y-6">
        <div className="space-y-3">
          <Label className="text-sm font-semibold text-slate-700 flex items-center gap-2">
            <MapPin className="w-4 h-4 text-indigo-500" />
            Distancia máxima: {filters.maxDistance || 10} km
          </Label>
          <Slider
            value={[filters.maxDistance || 10]}
            onValueChange={handleDistanceChange}
            max={50}
            min={1}
            step={1}
            className="w-full"
          />
        </div>

        <div className="space-y-3">
          <Label className="text-sm font-semibold text-slate-700 flex items-center gap-2">
            <Euro className="w-4 h-4 text-amber-500" />
            Presupuesto máximo: {filters.maxBudget || 50}€
          </Label>
          <Slider
            value={[filters.maxBudget || 50]}
            onValueChange={handleBudgetChange}
            max={200}
            min={0}
            step={5}
            className="w-full"
          />
        </div>

        <div className="space-y-3">
          <Label className="text-sm font-semibold text-slate-700 flex items-center gap-2">
            <Heart className="w-4 h-4 text-purple-500" />
            ¿Qué te interesa?
          </Label>

          <div className="flex flex-wrap gap-2">
            {culturalInterests.map((interest) => {
              const active = filters.interests?.includes(interest);
              return (
                <Badge
                  key={interest}
                  onClick={() => toggleInterest(interest)}
                  className={`cursor-pointer select-none ${
                    active ? "bg-indigo-600 text-white" : "bg-slate-100 text-slate-700"
                  }`}
                >
                  {interest}
                </Badge>
              );
            })}
          </div>
        </div>

        <div className="pt-2">
          <Button
            onClick={onSearch}
            className="w-full bg-gradient-to-r from-indigo-600 to-purple-600 hover:from-indigo-700 hover:to-purple-700"
          >
            Buscar
          </Button>
        </div>
      </CardContent>
    </Card>
  );
}
